                    public void fieldClicked() {
            if (fIsOpen) {
                return;
            } else if (cfIsBomb) {
                fStopDESC = true;
            } else if (!fStopDESC) {
                nGoodfields--;
            }
            unsetFlag();
            fIsOpen = true;
            cText.setVisible(true);
            cBorder.setFill(setTXTcolor(cText));
            if (cfIsBomb) {
                fDisplayWL = true;
                if (!fWon) {
                    setBomb();
                    displayWinLoseMessage();
                } else {
                    unsetBomb();
                }
                for (int i = 0; i < nY_felder; i++) {
                    for (int j = 0; j < nX_felder; j++) {
                        grid[j][i].fieldClicked();
                    }
                }
                unsetAllFlags();
                return;
            }
            if (cText.getText().isEmpty()) {
                getFieldsAround(this).forEach(Feld::fieldClicked);
            }

            if (nGoodfields == 0) {
                fDisplayWL = true;
                fWon = true;
                displayWinLoseMessage();
                for (int i = 0; i < nY_felder; i++) {
                    for (int j = 0; j < nX_felder; j++) {
                        grid[j][i].fieldClicked();
                    }
                }
                unsetAllFlags();
            }
        }
